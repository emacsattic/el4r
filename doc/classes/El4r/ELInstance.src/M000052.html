    def el4r_rubyproc_to_lambda(attrs = nil, &block)
      forms =
        ["el4r-lambda-for-rubyproc \"#{el4r_rubyobj_stock.obj2id(block)}\""]
      if attrs
        docstring = attrs[:docstring]
        forms << el4r_ruby2lisp(docstring) if docstring
        interactive = attrs[:interactive]
        if interactive
          case interactive
          when true; forms << "'(interactive)"
          else; forms << "'(interactive #{el4r_ruby2lisp(interactive)})"
          end
        end
      end
      "(#{forms.join(' ')})"
    end
